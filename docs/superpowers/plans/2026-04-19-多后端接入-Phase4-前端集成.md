> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

添加一个辅助函数来处理 Hermes stream：

```rust
async fn handle_hermes_stream(
    app_handle: AppHandle,
    mut stream: Box<dyn Stream>,
) {
    let mut full_text = String::new();

    loop {
        match stream.poll_next() {
            Some(Ok(response)) => {
                // 累积文本
                if !response.text_delta.is_empty() {
                    full_text.push_str(&response.text_delta);
                }

                // 如果有工具事件，已在 Task 1 中通过 tool_event_tx 发送

                // 如果是最终响应
                if response.is_final {
                    let payload = ResponseBubblePayload {
                        text: full_text.clone(),
                    };
                    let _ = app_handle.emit("update-response-bubble", payload);
                    break;
                }
            }
            Some(Err(e)) => {
                log::error!("[Hermes] stream error: {}", e);
                break;
            }
            None => {
                log::info!("[Hermes] stream ended");
                break;
            }
        }
    }
}
```

- [ ] **Step 4: 修改 send_to_hermes 使用 stream 处理**

```rust
#[tauri::command]
async fn send_to_hermes(message: String, app: AppHandle) -> Result<(), String> {
    let state = app.state::<AppState>();
    let manager = state.backend_manager.lock().map_err(|e| e.to_string())?;

    let clients = /* 获取 clients */;
    let hermes_client = clients.get("hermes").ok_or("Hermes client not found")?;

    let stream = hermes_client.send_message(&message).map_err(|e| e.to_string())?;

    // 启动任务处理 stream
    let app_for_task = app.clone();
    tokio::spawn(async move {
        handle_hermes_stream(app_for_task, stream).await;
    });

    log::info!("[Hermes] send_message: {}", message);
    Ok(())
}
```

```rust
.on_menu_event(move |app, event| {
    match event.id().as_ref() {
        "quit" => {
            log::info!("[Tray] Quit clicked");
            app.exit(0);
        }
        "restart" => {
            log::info!("[Tray] Restart clicked");
            let app_handle = app.clone();
            std::thread::spawn(move || {
                std::thread::sleep(std::time::Duration::from_millis(100));
                app_handle.restart();
            });
        }
        "options" => {
            log::info!("[Tray] Options clicked");
            if let Err(err) = show_options_window(app) {
                log::error!("[Tray] Failed to open options window: {}", err);
            }
        }
        "backend_openclaw" => {
            log::info!("[Tray] Switch to OpenClaw");
            let state = app.state::<AppState>();
            if let Ok(manager) = state.backend_manager.lock() {
                if let Err(e) = manager.switch("openclaw") {
                    log::error!("[Tray] Failed to switch to OpenClaw: {}", e);
                } else {
                    let _ = app.emit("backend-switched", "openclaw");
                }
            }
        }
        "backend_hermes" => {
            log::info!("[Tray] Switch to Hermes");
            let state = app.state::<AppState>();
            if let Ok(manager) = state.backend_manager.lock() {
                if let Err(e) = manager.switch("hermes") {
                    log::error!("[Tray] Failed to switch to Hermes: {}", e);
                } else {
                    let _ = app.emit("backend-switched", "hermes");
                }
            }
        }
        "check_status" => {
            log::info!("[Tray] Check status - opening options");
            if let Err(err) = show_options_window(app) {
                log::error!("[Tray] Failed to open options window: {}", err);
            }
        }
        _ => {}
    }
})
```

- [ ] **Step 4: 添加后端状态检查和切换逻辑**

在 `applyButton.addEventListener` 之后添加：

```typescript
import { invoke } from '@tauri-apps/api/core';
import { listen } from '@tauri-apps/api/event';

// 后端状态检查
async function checkBackendStatus() {
  try {
    const health = await invoke<BackendHealth>('check_backend_health');
    const statusEl = document.getElementById(`${health.backend}-status`)?.querySelector('.backend-indicator');
    if (statusEl) {
      statusEl.className = `backend-indicator ${health.available ? 'running' : 'stopped'}`;
      statusEl.textContent = health.available ? '运行中' : '未运行';
    }
    return health;
  } catch (e) {
    console.error('Failed to check backend health:', e);
    return null;
  }
}

// 获取当前后端
async function getCurrentBackend() {
  try {
    const backend = await invoke<string | null>('get_current_backend');
    const nameEl = document.getElementById('current-backend-name');
    if (nameEl && backend) {
      nameEl.textContent = backend === 'openclaw' ? 'OpenClaw' : 'Hermes';
    }
    return backend;
  } catch (e) {
    console.error('Failed to get current backend:', e);
    return null;
  }
}

// 切换后端
async function switchBackend(backend: string) {
  try {
    await invoke('switch_backend', { backend });
    await getCurrentBackend();
    await checkBackendStatus();
  } catch (e) {
    console.error('Failed to switch backend:', e);
  }
}

// 绑定按钮事件
document.getElementById('switch-to-openclaw')?.addEventListener('click', () => switchBackend('openclaw'));
document.getElementById('switch-to-hermes')?.addEventListener('click', () => switchBackend('hermes'));

// 初始化状态
checkBackendStatus();
getCurrentBackend();

// 监听后端切换事件
listen('backend-switched', (event) => {
  const nameEl = document.getElementById('current-backend-name');
  if (nameEl) {
    nameEl.textContent = event.payload === 'openclaw' ? 'OpenClaw' : 'Hermes';
  }
  checkBackendStatus();
});
```
