function showToast(message: string, isSuccess: boolean) {
  const toast = document.getElementById('toast');

    toast.textContent = isSuccess ? `✓ ${message}` : `✗ ${message}`;

- [ ] **Step 2: 重写 switchBackend 函数为 verifySwitch**

```typescript
async function switchBackend(backend: string) {
  try {
    // 1. 执行切换
    await invoke('switch_backend', { backend });

    // 2. 双重验证
    const result = await verifySwitch(backend);

    // 3. 显示结果
    showToast(result.message, result.success);

    // 4. 更新 UI
    await getCurrentBackend();
    await checkBackendStatus();

    return result;
  } catch (e) {
    console.error('Failed to switch backend:', e);
    showToast('切换失败：执行异常', false);
    return { success: false, message: '切换失败：执行异常' };
  }
}

async function verifySwitch(targetBackend: string): Promise<{success: boolean, message: string}> {
  // 验证1：检查当前后端是否切换
  const current = await invoke<string | null>('get_current_backend');
  if (current !== targetBackend) {
    return { success: false, message: '切换失败：状态异常' };
  }

  // 验证2：检查目标后端是否可用
  const health = await invoke<BackendHealth>('check_backend_health');
  if (!health.available) {
    const backendName = targetBackend === 'hermes' ? 'Hermes' : 'OpenClaw';
    return { success: false, message: `切换失败：${backendName} 未响应` };
  }

  const backendName = targetBackend === 'hermes' ? 'Hermes' : 'OpenClaw';
  return { success: true, message: `已切换到 ${backendName}` };
}
```
