# 前端状态机与动画性能优化

## 一、XState 与 Vue 集成

### 为什么用 XState？

XState 是 **StateChart** 的 JavaScript 实现，比手写 FSM 更强大：

- **层级状态** (Hierarchical)
- **并行状态** (Parallel)
- **历史状态** (History)
- **可观测性** (Inspect)

```typescript
// 安装
npm install xstate @xstate/vue
```

### 桌面宠物的状态机模型

```typescript
import { createMachine, assign } from 'xstate';
import { useMachine } from '@xstate/vue';

// 定义状态机
const petMachine = createMachine({
    id: 'pet',
    initial: 'idle',
    context: {
        energy: 100,
        hunger: 0,
        mood: 'happy'
    },
    states: {
        idle: {
            on: {
                CLICK: 'alert',
                DRAG_START: 'dragging',
                TIMEOUT: { target: 'idle', actions: 'decreaseEnergy' }
            },
            entry: 'playIdleAnimation',
            exit: 'stopIdleAnimation'
        },
        
        alert: {
            on: {
                DONE: 'idle'
            },
            entry: 'playShockAnimation'
        },
        
        dragging: {
            type: 'parallel',
            states: {
                movement: {
                    initial: 'moving',
                    states: {
                        moving: {
                            on: { DRAG_END: 'stopped' }
                        },
                        stopped: {
                            on: { DRAG_START: 'moving' }
                        }
                    }
                },
                animation: {
                    initial: 'dragging',
                    states: {
                        dragging: {
                            on: { VELOCITY_HIGH: 'throwing' }
                        },
                        throwing: {
                            on: { LANDED: 'dragging' }
                        }
                    }
                }
            },
            on: {
                DRAG_END: {
                    target: 'idle',
                    actions: ['applyVelocity', 'checkLanding']
                }
            }
        },
        
        eating: {
            on: {
                FINISH: 'idle'
            },
            entry: ['playEatAnimation', 'increaseEnergy'],
            exit: 'stopEatAnimation'
        }
    },
    
    on: {
        UPDATE_POSITION: {
            actions: 'updatePosition'
        }
    }
}, {
    actions: {
        playIdleAnimation: () => { /* 调用精灵图 */ },
        decreaseEnergy: assign({ energy: (ctx) => ctx.energy - 1 }),
        applyVelocity: (ctx, event) => {
            if (event.type === 'DRAG_END') {
                // 施加速度
            }
        }
    }
});
```

### Vue 组件中使用

```vue
<template>
  <div @click="send('CLICK')">
    <PetSprite :animation="animationName" />
    <Bubble v-if="current.matches('alert')" message="!" />
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import { useMachine } from '@xstate/vue';
import { petMachine } from './petMachine';

const { snapshot, send } = useMachine(petMachine);

const animationName = computed(() => {
    const state = snapshot.value;
    
    if (state.matches('idle')) return 'idle';
    if (state.matches({ alert: 'shock' })) return 'shock';
    if (state.matches('dragging')) return 'dragging';
    if (state.matches('eating')) return 'eating';
    
    return 'idle';
});
</script>
```

## 二、OffscreenCanvas + Web Worker 渲染

### 核心问题

Canvas 渲染在主线程，当主线程繁忙（大量计算、DOM 操作）时，动画会掉帧。

### 解决方案

将渲染工作移到 Web Worker：

```
┌─────────────────┐     ┌─────────────────┐
│   Main Thread   │     │   Web Worker    │
│                 │     │                 │
│  - 物理计算     │ ──▶ │  - Canvas 渲染  │
│  - 状态机       │     │  - 精灵图绘制   │
│  - DOM 更新     │ ◀── │  - 帧计算       │
└─────────────────┘     └─────────────────┘
```

### Worker 渲染器

```typescript
// worker/render.worker.ts
const canvas = new OffscreenCanvas(180, 180);
const ctx = canvas.getContext('2d')!;

let spriteImage: ImageBitmap | null = null;
let currentFrame = 0;
let frameCount = 16;
let cols = 4;
let frameWidth = 180;
let frameHeight = 180;

// 接收精灵图
self.onmessage = async (e) => {
    const { type, data } = e.data;
    
    switch (type) {
        case 'LOAD_SPRITE':
            const blob = await fetch(data.url).then(r => r.blob());
            spriteImage = await createImageBitmap(blob);
            frameCount = data.frameCount || 16;
            cols = data.cols || 4;
            frameWidth = data.frameWidth || 180;
            frameHeight = data.frameHeight || 180;
            break;
            
        case 'RENDER':
            drawFrame(currentFrame);
            // 发送回主线程
            const bitmap = canvas.transferToImageBitmap();
            self.postMessage({ type: 'FRAME', bitmap }, [bitmap]);
            break;
            
        case 'NEXT_FRAME':
            currentFrame = (currentFrame + 1) % frameCount;
            break;
    }
};

function drawFrame(frame: number) {
    if (!spriteImage) return;
    
    ctx.clearRect(0, 0, 180, 180);
    
    const col = frame % cols;
    const row = Math.floor(frame / cols);
    const sx = col * frameWidth;
    const sy = row * frameHeight;
    
    ctx.drawImage(
        spriteImage,
        sx, sy, frameWidth, frameHeight,
        0, 0, 180, 180
    );
}
```

### 主线程集成

```typescript
// composables/useOffscreenRender.ts
import { ref, onMounted, onUnmounted } from 'vue';

export function useOffscreenRender(width: number, height: number) {
    const canvas = ref<HTMLCanvasElement | null>(null);
    const worker = ref<Worker | null>(null);
    const currentBitmap = ref<ImageBitmap | null>(null);
    let animationFrameId: number;
    
    onMounted(() => {
        // 创建 Worker
        worker.value = new Worker(
            new URL('./worker/render.worker.ts', import.meta.url),
            { type: 'module' }
        );
        
        worker.value.onmessage = (e) => {
            if (e.data.type === 'FRAME') {
                currentBitmap.value = e.data.bitmap;
                // 绘制到可见 Canvas
                if (canvas.value) {
                    const ctx = canvas.value.getContext('2d')!;
                    ctx.drawImage(e.data.bitmap, 0, 0);
                }
            }
        };
        
        // 加载精灵图
        worker.value.postMessage({
            type: 'LOAD_SPRITE',
            data: {
                url: '/anim/idle.png',
                frameCount: 16,
                cols: 4,
                frameWidth: 180,
                frameHeight: 180
            }
        });
        
        // 开始渲染循环
        function render() {
            worker.value?.postMessage({ type: 'RENDER' });
            animationFrameId = requestAnimationFrame(render);
        }
        render();
    });
    
    function setAnimation(name: string, config: AnimationConfig) {
        worker.value?.postMessage({
            type: 'LOAD_SPRITE',
            data: {
                url: config.imagePath,
                frameCount: config.totalFrames,
                cols: config.cols,
                frameWidth: config.frameWidth,
                frameHeight: config.frameHeight
            }
        });
    }
    
    onUnmounted(() => {
        cancelAnimationFrame(animationFrameId);
        worker.value?.terminate();
    });
    
    return { canvas, currentBitmap, setAnimation };
}
```

### 性能对比

| 方案 | CPU 密集时帧率 | 优势 | 劣势 |
|------|--------------|------|------|
| 主线程 Canvas | 10-15fps | 简单 | 容易被阻塞 |
| OffscreenCanvas | 55-60fps | 流畅 | 实现复杂 |
| Web Worker + OffscreenCanvas | 60fps | 完全独立 | 数据传递开销 |

## 三、TypeScript 类型生成 (TS-RS)

### 问题

Rust 端和 TypeScript 端的类型需要手动同步，容易出错：

```rust
// Rust
#[derive(Serialize, Deserialize)]
pub enum AnimState {
    Idle,
    Shock,
    Working,
}
```

```typescript
// TypeScript - 手动同步，容易写错！
type AnimState = 'Idle' | 'Shock' | 'Working';
```

### TS-RS 方案

TS-RS 可以从 Rust 代码自动生成 TypeScript 类型：

```rust
// Cargo.toml
[dependencies]
ts-rs = "7"
```

```rust
use ts_rs::TS;

#[derive(Serialize, Deserialize, TS)]
#[ts(export)]
pub enum AnimState {
    Idle,
    Shock,
    Working,
}

#[derive(Serialize, Deserialize, TS)]
#[ts(export)]
pub struct AnimationConfig {
    name: String,
    total_frames: u32,
    interval: u32,
}
```

运行 `cargo build` 后，TS-RS 会生成 `.ts` 文件：

```typescript
// generated/anim_state.ts
export enum AnimState {
    Idle = 'Idle',
    Shock = 'Shock',
    Working = 'Working',
}

export interface AnimationConfig {
    name: string;
    total_frames: number;
    interval: number;
}
```

### 在 Tauri 中使用

```typescript
import { AnimState, AnimationConfig } from '../generated/anim_state';

// 自动类型检查
function switchAnimation(state: AnimState): Promise<void> {
    return invoke('switch_animation', { state });
}

// 完整类型提示
const config: AnimationConfig = {
    name: 'idle',
    total_frames: 16,
    interval: 100
};
```
