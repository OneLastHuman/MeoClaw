# 物理系统与动效集成参考

## 概述

桌面宠物的"真实感"很大程度上来自于物理模拟——重力感、惯性、碰撞等。纯 CSS/精灵图动画只能实现预定义的帧序列，而物理系统能实现更自然的动态效果。

## 物理引擎选择

| 引擎 | 特点 | 适用场景 |
|------|------|----------|
| **Matter.js** | 2D 物理，API 友好，文档完善 | 桌面宠物、网页游戏 |
| **Planck.js** | 2D 物理，Box2D 的 JavaScript 移植版 | 精确物理模拟 |
| **Dynamics.js** | 轻量，仅 9 种动效 | UI 交互反馈 |
| **react-spring** | 弹性动画，React 友好 | UI 动画 |

对于桌面宠物应用，**Matter.js** 是最合适的选择。

## Matter.js 核心概念

### 物理世界结构

```
Engine
├── World (物理世界)
│   ├── Bodies (刚体)
│   │   ├── Rectangle (矩形)
│   │   ├── Circle (圆形)
│   │   └── Polygon (多边形)
│   ├── Constraint (约束/关节)
│   └── Composite (复合体)
├── Runner (物理步进)
└── Render (渲染器，可选)
```

### 关键 API

```javascript
import Matter from 'matter-js';

const { Engine, World, Bodies, Body, Runner } = Matter;

// 1. 创建引擎
const engine = Engine.create({
    gravity: { x: 0, y: 1 }  // 重力
});

// 2. 创建刚体
const cat = Bodies.rectangle(400, 200, 180, 180, {
    friction: 0.3,
    restitution: 0.5,  // 弹性
    label: 'cat'
});

// 3. 添加到世界
World.add(engine.world, cat);

// 4. 创建步进器
const runner = Runner.create();
Runner.run(runner, engine);

// 5. 获取位置/角度
function update() {
    const { x, y } = cat.position;
    const angle = cat.angle;
    // 更新宠物位置
}
```

### 集成示例

```typescript
// composables/usePhysics.ts
import { ref, onMounted, onUnmounted } from 'vue';
import Matter from 'matter-js';

export function usePhysics() {
    const engine = ref<Matter.Engine | null>(null);
    const runner = ref<Matter.Runner | null>(null);
    const catBody = ref<Matter.Body | null>(null);
    
    // 宠物物理配置
    const CAT_CONFIG = {
        x: 100,
        y: 100,
        width: 180,
        height: 180,
        friction: 0.3,
        restitution: 0.6,  // 弹跳
        density: 0.001,    // 质量
    };
    
    function init() {
        // 创建引擎
        engine.value = Matter.Engine.create({
            gravity: { x: 0, y: 0.5 }
        });
        
        // 创建宠物刚体
        catBody.value = Matter.Bodies.rectangle(
            CAT_CONFIG.x,
            CAT_CONFIG.y,
            CAT_CONFIG.width,
            CAT_CONFIG.height,
            {
                friction: CAT_CONFIG.friction,
                restitution: CAT_CONFIG.restitution,
                density: CAT_CONFIG.density,
                label: 'pet',
                // 物理碰撞边界
                chamfer: { radius: 20 },  // 圆角
            }
        );
        
        Matter.World.add(engine.value.world, catBody.value);
        
        // 创建步进器
        runner.value = Matter.Runner.create();
        Matter.Runner.run(runner.value, engine.value);
        
        // 监听更新
        Matter.Events.on(engine.value, 'afterUpdate', onPhysicsUpdate);
    }
    
    function onPhysicsUpdate() {
        if (!catBody.value) return;
        
        const { x, y } = catBody.value.position;
        const angle = catBody.value.angle;
        
        // 发送到 Vue 组件更新渲染
        updatePetPosition(x, y, angle);
    }
    
    // 施力
    function applyForce(forceX: number, forceY: number) {
        if (!catBody.value) return;
        Matter.Body.applyForce(catBody.value, catBody.value.position, {
            x: forceX,
            y: forceY
        });
    }
    
    // 拖拽结束时的惯性
    function throwPet(velocityX: number, velocityY: number) {
        if (!catBody.value) return;
        Matter.Body.setVelocity(catBody.value, {
            x: velocityX,
            y: velocityY
        });
    }
    
    function cleanup() {
        if (runner.value) Matter.Runner.stop(runner.value);
        if (engine.value) Matter.Engine.clear(engine.value);
    }
    
    onMounted(init);
    onUnmounted(cleanup);
    
    return {
        engine,
        catBody,
        applyForce,
        throwPet
    };
}
```

### 与精灵图同步

```typescript
// composables/usePhysicsPet.ts
import { ref, computed } from 'vue';
import { usePhysics } from './usePhysics';

export function usePhysicsPet(canvasRef: Ref<HTMLCanvasElement | null>) {
    const { catBody, applyForce, throwPet } = usePhysics();
    
    const petPosition = ref({ x: 0, y: 0, angle: 0 });
    const isPhysicsActive = ref(true);
    
    // 监听物理更新
    watch(catBody, (body) => {
        if (!body) return;
        
        Matter.Events.on(body.engine!, 'afterUpdate', () => {
            petPosition.value = {
                x: body.position.x - 90,  // 中心点补偿
                y: body.position.y - 90,
                angle: body.angle
            };
        });
    });
    
    // 鼠标拖拽交互
    function onDragStart(e: MouseEvent) {
        if (!catBody.value || !canvasRef.value) return;
        
        const rect = canvasRef.value.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        // 检查是否在宠物范围内
        if (Matter.Bounds.contains(catBody.value.bounds, { x, y })) {
            isPhysicsActive.value = false;
            // 开始拖拽
            Matter.Body.setStatic(catBody.value, true);
        }
    }
    
    function onDragMove(e: MouseEvent) {
        if (!catBody.value || !canvasRef.value) return;
        if (isPhysicsActive.value) return;
        
        const rect = canvasRef.value.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        Matter.Body.setPosition(catBody.value, { x, y });
        petPosition.value = { x, y, angle: 0 };
    }
    
    function onDragEnd(e: MouseEvent) {
        if (!catBody.value) return;
        if (isPhysicsActive.value) return;
        
        // 计算抛出速度
        const velocityX = (e.movementX || 0) * 0.5;
        const velocityY = (e.movementY || 0) * 0.5;
        
        // 恢复物理
        Matter.Body.setStatic(catBody.value, false);
        throwPet(velocityX, velocityY);
        
        isPhysicsActive.value = true;
    }
    
    return {
        petPosition,
        isPhysicsActive,
        onDragStart,
        onDragMove,
        onDragEnd,
        applyForce
    };
}
```

## 动效库对比

### 1. Matter.js — 完整物理引擎

```javascript
// 物理模拟
const engine = Matter.Engine.create();
const cat = Matter.Bodies.circle(400, 200, 50, {
    restitution: 0.8  // 弹性
});
Matter.World.add(engine.world, cat);
```

### 2. Dynamics.js — 轻量弹簧动效

适合 UI 交互反馈（按钮点击、窗口拖拽）：

```javascript
import dynamics from 'dynamics.js';

// 弹跳效果
dynamics.animate(element, {
    translateY: -100,
    opacity: 1
}, {
    type: dynamics.spring,
    friction: 50,
    change: () => render()
});

// 弹性衰减
dynamics.animate(element, {
    scale: 1
}, {
    type: dynamics.bounce,
    frequency: 15,
    change: () => render()
});
```

### 3. react-spring — 现代弹性动画

```tsx
import { useSpring, animated } from '@react-spring/web';

function Pet({ isThrown }: { isThrown: boolean }) {
    const [spring, api] = useSpring(() => ({
        x: 0,
        y: 0,
        rotate: 0,
        config: { tension: 300, friction: 10 }
    }));
    
    // 投掷时触发动画
    useEffect(() => {
        if (isThrown) {
            api.start({
                x: throwX,
                y: throwY,
                rotate: 360
            });
        }
    }, [isThrown]);
    
    return (
        <animated.div
            style={{
                transform: spring.x.to(x => `translateX(${x}px)`),
            }}
        />
    );
}
```

## 性能优化

### 1. 限制更新频率

```typescript
// 不需要每帧更新物理状态
Matter.Events.on(engine, 'afterUpdate', () => {
    // 限制为 30fps 更新渲染
    if (Date.now() - lastRenderTime < 33) return;
    lastRenderTime = Date.now();
    updateVisual();

```

### 2. 使用 Sleeping

```javascript
const cat = Bodies.rectangle(400, 200, 180, 180, {
    sleepThreshold: 60  // 静止多久后进入睡眠
});
```

### 3. 简化碰撞检测

```typescript
// 只检测与地面的碰撞
Matter.Events.on(engine, 'collisionStart', (event) => {
    const ground = Matter.Composite.allBodies(engine.world)
        .find(b => b.label === 'ground');
    
    event.pairs = event.pairs.filter(
        p => p.bodyA === ground || p.bodyB === ground
    );
});
```

## 与 Tauri 窗口坐标同步

物理引擎在 Canvas 内部模拟，窗口移动需要同步：

```typescript
// 物理引擎内部坐标
const PHYSICS_SCALE = 1;  // 1:1 映射

// 物理位置 -> 窗口位置
function physicsToWindow(x: number, y: number) {
    return {
        x: x * PHYSICS_SCALE,
        y: y * PHYSICS_SCALE
    };
}

// 监听 Tauri 窗口移动事件
import { getCurrentWindow } from '@tauri-apps/api/window';

const win = getCurrentWindow();
win.onMoved(({ payload }) => {
    // 通知物理引擎窗口位置变化
    emit('window:moved', payload.position);
});
```
